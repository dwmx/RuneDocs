// ##################################
// # UWindow.UWindowServerQuery
// ##################################

class UWindowServerQuery extends UdpLink;

var WindowConsole   Server;

var IpAddr              ServerIPAddr;

var int                 PingAttempts;
var int                 AttemptNumber;
var int                 BindAttempts;
var name                QueryState;

var config int          MaxBindAttempts;
var config int          BindRetryTime;
var config int          PingTimeout;

var bool ImDone;

function ValidateServer()
{
    if(Server.ServerPing != Self)
    {
        Log("ORPHANED: "$Self);
        Destroy();
    }
}

function string ParseReply(string Text, string Key)
{
    local int i;
    local string Temp;

    i=InStr(Text, "\\"$Key$"\\");
    Temp = Mid(Text, i + Len(Key) + 2);
    return Left(Temp, InStr(Temp, "\\"));
}

state GetVer
{
    event ReceivedText(IpAddr Addr, string Text)
    {
        local string Temp;
        local float ElapsedTime;

        // Make sure this packet really is for us.
//		Temp = IpAddrToString(Addr);
//		if(Server.IP != Left(Temp, InStr(Temp, ":")))
//			return;

        ValidateServer();

        Temp = ParseReply(Text, "EngineVersion");
        if(Left(Temp, 3) ~= "108")
        {
                Log("EngineVersion "$Temp, 'Progress');
                QueryState = 'GetInfo';
                SetTimer(0.0, False);
                if(ImDone) return;
                GotoState(QueryState);
        }
        else
        {
                Log("EngineVersion "$Temp, 'Progress');
                ImDone = True;
            Server.PingDone(False);
        }
    }
    event Timer()
    {
        ValidateServer();
            if (ImDone) return;
        if(AttemptNumber < PingAttempts)
        {
            Log("Timed out getting engine version replies.  Attempt "$AttemptNumber, 'Progress');
            AttemptNumber++;
            GotoState('ReRoute');
        }
        else
        {
                ImDone = true;
            Server.PingDone(False);
            Log("Timed out getting redirection replies.  Giving Up", 'Progress');
        }
    }
Begin:
    ValidateServer();
    if (!ImDone)
        {
            SendText( ServerIPAddr, "\\level_property\\EngineVersion\\" );
            log("Asked engine version from IP Address'"$IpAddrToString(ServerIPAddr)$"'");
    }
    if (!ImDone) SetTimer(PingTimeout, False);
}

function StartQuery(name S, int InPingAttempts)
{
    QueryState = S;
    ValidateServer();
    if(ImDone) return;
    ServerIPAddr.Port = Server.TryServer.Port + 1;
    GotoState('Resolving');
    PingAttempts=InPingAttempts;
    AttemptNumber=1;
}

function Resolved( IpAddr Addr )
{
    ServerIPAddr.Addr = Addr.Addr;
    GotoState('Binding');
}

function bool GetNextValue(string In, out string Out, out string Result)
{
    local int i;
    local bool bFoundStart;

    Result = "";
    bFoundStart = False;

    for(i=0;i<Len(In);i++)
    {
        if(bFoundStart)
        {
            if(Mid(In, i, 1) == "\\")
            {
                Out = Right(In, Len(In) - i);
                return True;
            }
            else
            {
                Result = Result $ Mid(In, i, 1);
            }
        }
        else
        {
            if(Mid(In, i, 1) == "\\")
            {
                bFoundStart = True;
            }
        }
    }

    return False;
}

state Binding
{
Begin:
    if( BindPort(2000, true) == 0 )
    {
        Log("UWindowServerQuery: Port failed to bind.  Attempt "$BindAttempts, 'Progress');
        BindAttempts++;

        ValidateServer();
        if(BindAttempts == MaxBindAttempts)
            Server.PingDone(False);
        else
            GotoState('BindFailed');
    }
    else
    {
        GotoState(QueryState);
    }
}


state BindFailed
{
    event Timer()
    {
        GotoState('Binding');
    }

Begin:
    SetTimer(BindRetryTime, False);
}


state GetInfo
{
    event ReceivedText( IpAddr Addr, string Text )
    {
        local string Value;
        local string In;
        local string Out;
        local byte ID;
        local bool bOK;
        
        ValidateServer();
        
        if(ImDone) return;

        In = Text;
        do
        {
            bOK = GetNextValue(In, Out, Value);
            In = Out;
            if(Value ~= "packagelist")
            {
                bOK = GetNextValue(In, Out, Value);
                ValidateServer();
//				if (Value == "") Server.PingDone(False);
                Server.TryServer.PackageList = Value;
            }
            else if(Value == "downloadurls")
            {
                bOK = GetNextValue(In, Out, Value);
                ValidateServer();
//				if (Value == "") Server.PingDone(False);
                Server.TryServer.DownloadUrl = Value;
                bOK = false;
                if (!ImDone) Server.PingDone(True);
                ImDone = true;
                        SetTimer(0.0, False);
                Destroy();
            }
            else if(Value == "final")
            {
                if (!ImDone) Server.PingDone(False);
                return;
            }
        } until(!bOK);
    }
    event Timer()
    {
            if (ImDone) return;
        if(AttemptNumber < PingAttempts)
        {
            Log("Timed out getting redirection replies.  Attempt "$AttemptNumber, 'Progress');
            AttemptNumber++;
            GotoState('ReRoute');
        }
        else
        {
                ImDone = true;
            Server.PingDone(False);
            Log("Timed out getting redirection replies.  Giving Up", 'Progress');
        }
    }
Begin:
    ValidateServer();
    if (!ImDone)
        {
                SendText( ServerIPAddr, "\\packages\\" );
            log("Sent: \\packages\\ to IP Address'"$IpAddrToString(ServerIPAddr)$"'");
    }
    if (!ImDone) SetTimer(PingTimeout, False);
}

state ReRoute
{
Begin:
      GotoState(QueryState);
}

state Resolving
{
Begin:
    Resolve( Server.TryServer.IP );
}

defaultproperties
{
     MaxBindAttempts=2
     BindRetryTime=2
     PingTimeout=5
}
