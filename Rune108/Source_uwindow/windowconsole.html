struct MyServerInfo
{
    var string URL; // the command to join
    var bool bRelative; // travel relative

    var int Port; // Game Port
    var string LevelName; // Possibly Map name

function bool ConsoleCommand(coerce string S)
{
    local string T,Rec;
    local int i;
    T = S;
    if (Left(S,6) ~= "Cancel" || Left(S,10) ~= "Disconnect")
    {
        if (ServerPing != None)
        {
                ServerPing.ImDone = true;
                ServerPing.Destroy();
                ServerPing = None;
        }
    bPinging = False;
    redirectionOnProgress = 0;
    Rec = CurrentServer.URL;
    cleanLists();
    CurrentServer.URL = Rec;
    Super.ConsoleCommand(S);
    return true;
    }
    else if (Left(S,9) ~= "Reconnect")
    {
        if (ServerPing != None)
        {
                ServerPing.ImDone = true;
                ServerPing.Destroy();
                ServerPing = None;
        }
    bPinging = False;
    redirectionOnProgress = 0;
    Super.ConsoleCommand("Disconnect");
    T = CurrentServer.URL;
    cleanLists();
    CurrentServer.URL = T;
    Viewport.Actor.ClearProgressMessages();
    IsOpenUrl(T);
    return true;
    }
    else if (IsRedirectionCommand(T)) return true;
    return Super.ConsoleCommand(T);
}

function bool IsRedirectionCommand(out string S)
{
    if (IsOpenUrl(S)) return true;
    else if (IsRelativeTravel(S)) return true;
//    Log("Not a redirection command: "$S);
    return false;
}

function bool IsRelativeTravel(out string S)
{
    local string T;
    local string zURL, zIP, zLevelName, zEntryName;
    local int zPort;
    local UWindowServerQuery U;
    if (Left(S,8) ~= "RTravel ")
    {
        closeOldConnection();
        if (bPinging) ServerPing = None;
        TryServer.URL = S;
        TryServer.bRelative = true;
        T = Right(S,Len(S)-8);
        U = Viewport.Actor.GetEntryLevel().Spawn(class'UWindowServerQuery');
        if (U.ParseURL(T, zIP, zPort, TryServer.LevelName, zEntryName))
        {
//            log("Following  Info: URL '"$T$"', IP '"$zIP$"', Port '"$zPort$"', LevelName '"$TryServer.LevelName$"', EntryName '"$zEntryName$"'");
//            log("Was Interpreted: URL '"$T$"', IP '"$CurrentServer.DownloadUrl$"', Port '"$CurrentServer.Port$"', LevelName '"$TryServer.LevelName$"', EntryName '"$zEntryName$"'");
            log("Joining '"$CurrentServer.IP$":"$CurrentServer.Port$"'", 'Progress');
            U.Destroy();
            if (CurrentServer.DownloadUrl == "") return false;
            redirectionOnProgress = 1;
            downloadthis(CurrentServer.DownloadUrl, CurrentServer.Port, Left(TryServer.LevelName, (len(TryServer.LevelName)-4)) );
            return true;
        }
        U.Destroy();
    }
    return false;
}

function closeOldConnection()
{
        redirectionOnProgress = 0;
        proginfo = 0;
        if (ServerPing != None)
        {
                ServerPing.ImDone = true;
                ServerPing.Destroy();
                ServerPing = None;
        }
    bPinging = False;
}

function bool IsOpenUrl(out string S)
{
    local string T;
    local UWindowServerQuery U;
    if (Left(S,6) ~= "start ")
    {
        if (bPinging) ServerPing = None;
        closeOldConnection();
        LOG("START! '"$s$"'", 'Progress');
        TryServer.URL = S;
        T = Right(S,Len(S)-6);
        U = Viewport.Actor.GetEntryLevel().Spawn(class'UWindowServerQuery');
        if (U.ParseURL(T, TryServer.IP, TryServer.Port, TryServer.LevelName, TryServer.EntryName))
        {
//            log("Open: URL '"$TryServer.URL$"', IP '"$TryServer.IP$"', Port '"$TryServer.Port$"', LevelName '"$TryServer.LevelName$"', EntryName '"$TryServer.EntryName$"'");
            log(TryServer.URL, 'Progress');
            U.Destroy();
            if (TryServer.IP == "") return false;
            PingServer();
            return true;
        }
        U.Destroy();
    }
    return false;
}

function joinServer(bool bSuccess)
{
    if (bSuccess)
    {
//        if (!CurrentServer.bRelative) TryServer.URL = TryServer.URL$"?Ver=108";
        if (!TryServer.bRelative)
        {
                if(instr(TryServer.URL, "?Ver=108") == -1) TryServer.URL = TryServer.URL$"?Ver=108";
                CurrentServer.URL = TryServer.URL; // the command to join
            CurrentServer.IP = TryServer.IP; // IP where we want to join
            CurrentServer.Port = TryServer.Port; // Port
            CurrentServer.LevelName = TryServer.LevelName; // Possibly Map name
            CurrentServer.EntryName = TryServer.EntryName; // I hope these are the rest
            CurrentServer.DownloadUrl = TryServer.DownloadUrl;
            CurrentServer.PackageList = TryServer.PackageList;
        }
        CurrentServer.bRelative = TryServer.bRelative; // travel relative
        log("Redirected Downloads successfully complete, join the server " $ TryServer.IP $ " - " $ CurrentServer.IP, 'Progress');
    }
    else
    {
        log("Join the old fashioned way");
        CurrentServer.URL = TryServer.URL;
        CurrentServer.IP = TryServer.IP; // IP where we want to join
        CurrentServer.Port = TryServer.Port; // Port
    }
    HandleNormalJoin();
}

function cleanLists()
{
        CurrentServer.URL = "";
        TryServer.URL = ""; // the command to join
        CurrentServer.IP = "";
        TryServer.IP = ""; // IP where we want to join
        CurrentServer.Port = 0;
        TryServer.Port = 0; // Port
        CurrentServer.LevelName = "";
        TryServer.LevelName = ""; // Possibly Map name
        CurrentServer.EntryName = "";
        TryServer.EntryName = ""; // I hope these are the rest
        CurrentServer.DownloadUrl = "";
        TryServer.DownloadUrl = "";
        CurrentServer.PackageList = "";
        TryServer.PackageList = "";
        CurrentServer.bRelative = false;
        TryServer.bRelative = false; // travel relative
}

function PingServer()
{
    // Create the UdpLink to ping the server
    ServerPing = Viewport.Actor.GetEntryLevel().Spawn(class'UWindowServerQuery');
    ServerPing.Server = Self;
//	ServerPing.StartQuery('GetInfo', 2);
    ServerPing.StartQuery('GetVer', 2);
    Super.ConsoleCommand("Cancel");
    Viewport.Actor.ClearProgressMessages();
//	LOG("PINGSERVER!");
    bPinging = True;
}

function PingDone(bool Success)
{
    // Destroy the UdpLink
        if (ServerPing != None)
        {
                ServerPing.ImDone = true;
                ServerPing.Destroy();
                ServerPing = None;
        }
    bPinging = False;

//	LOG("PINGDONE! "$Success);

    if(!Success)
    {
            // Pinging failed for some reason
            // Join the 'normal' way
            proginfo = 0;
            JoinServer(false);
            return;
    }
//    log("downloadthis( "$TryServer.DownloadUrl$", "$TryServer.Port$", "$TryServer.PackageList$" )");
    redirectionOnProgress = 1;
    downloadthis(TryServer.DownloadUrl, TryServer.Port, TryServer.PackageList);
}

function HandleNormalJoin()
{
    local string S;
    S = CurrentServer.IP$":"$CurrentServer.Port;
    if (TryServer.bRelative && ( viewport.actor.Level.GetAddressURL() ~= S ))
    {
        Log("Join server, clienttravel: "$TryServer.LevelName$" Address: "$S);
        TryServer.bRelative = false;
        viewport.actor.ClientTravel(TryServer.LevelName, TRAVEL_Relative, false);
        return;
    }
    Log("Join server, normal: "$CurrentServer.URL);
    TryServer.bRelative = false;
    Super(Console).ConsoleCommand(CurrentServer.URL);
}

Function PostRender( canvas Canvas )
{
    local string S;
    local float X, Y;
    Super.PostRender(Canvas);
    if (bPinging) // "Querying the server... ";
    {
        S = "Querying the server... ";
        Canvas.TextSize(S,X,Y);
        X = Canvas.ClipX/2 - X/2;
        Y = Canvas.ClipY/2 - Y/2;
        Canvas.SetPos(X, Y);
        Canvas.Font = Canvas.MedFont;
        Canvas.DrawText(S, false);
    }
    else if (redirectionOnProgress == 1)
    {
//        Canvas.SetPos(Canvas.ClipX/2, Canvas.ClipY/3);
        Canvas.Font = Canvas.MedFont;
        S = OnProgress (progress, progsize);

        Canvas.TextSize(S,X,Y);
        X = Canvas.ClipX/2 - X/2;
        Y = Canvas.ClipY/2 - Y/2;
        Canvas.SetPos(X, Y);

        Canvas.DrawText(" "$S, false);
    }
    if ((proginfo & 0x0008) > 0) //....01XX
    {
            JoinServer(proginfo == 10);
            proginfo = 0;
            bPinging = false;
            if (ServerPing != None)
            {
                    ServerPing.ImDone = true;
                    ServerPing.Destroy();
                    ServerPing = None;
            }
            redirectionOnProgress = 0;
    }
}

Function String OnProgress (int mprogress, int mprogsize)
{
    if (proginfo < 1) return "";
    else if (proginfo == 2) return "Downloading "$mprogress$"kB/"$mprogsize$"kB "$statustext;
    else if (proginfo == 1) return "Connecting... ";
    else if (proginfo == 3) return "Installing... ";
    else if (proginfo == 5) return "Querying the server... ";
    else return ""$proginfo;
}

state UWindow
{
    event EndState()
    {
        Root.GetPlayerOwner().RendMap = 5;
    }

    event Tick( float Delta )
    {
        Global.Tick(Delta);
        if(Root != None)
            Root.DoTick(Delta);
    }

    event PostRender( canvas Canvas )
    {
        if( bTimeDemo )
        {   
            TimeDemoCalc();
            TimeDemoRender( Canvas );
        }

        if(Root != None)
            Root.bUWindowActive = True;
        RenderUWindow( Canvas );
    }

    event bool KeyType( EInputKey Key )
    {
        if (Root != None)
            Root.WindowEvent(WM_KeyType, None, MouseX, MouseY, Key);
        return True;
    }

    event bool KeyEvent( EInputKey Key, EInputAction Action, FLOAT Delta )
    {
        local byte k;
        k = Key;

        switch (Action)
        {
        case IST_Release:
            switch (k)
            {
            case EInputKey.IK_LeftMouse:
                if(Root != None) 
                    Root.WindowEvent(WM_LMouseUp, None, MouseX, MouseY, k);
                break;
            case EInputKey.IK_RightMouse:
                if(Root != None)
                    Root.WindowEvent(WM_RMouseUp, None, MouseX, MouseY, k);
                break;
            case EInputKey.IK_MiddleMouse:
                if(Root != None)
                    Root.WindowEvent(WM_MMouseUp, None, MouseX, MouseY, k);
                break;
            default:
                if(Root != None)
                    Root.WindowEvent(WM_KeyUp, None, MouseX, MouseY, k);
                break;
            }
            break;

        case IST_Press:
            switch (k)
            {
            case EInputKey.IK_F9:   // Screenshot
                return Global.KeyEvent(Key, Action, Delta);
                break;
            case ConsoleKey:
                if (bShowConsole)
                {
                    HideConsole();
                    if(bQuickKeyEnable)
                        CloseUWindow();
                }
                else
                {
                    if(Root.bAllowConsole)
                        ShowConsole();
                    else
                        Root.WindowEvent(WM_KeyDown, None, MouseX, MouseY, k);
                }
                break;
            case EInputKey.IK_Escape:
                if(Root != None)
                    Root.CloseActiveWindow();
                break;
            case EInputKey.IK_LeftMouse:
                if(Root != None)
                    Root.WindowEvent(WM_LMouseDown, None, MouseX, MouseY, k);
                break;
            case EInputKey.IK_RightMouse:
                if(Root != None)
                    Root.WindowEvent(WM_RMouseDown, None, MouseX, MouseY, k);
                break;
            case EInputKey.IK_MiddleMouse:
                if(Root != None)
                    Root.WindowEvent(WM_MMouseDown, None, MouseX, MouseY, k);
                break;
            default:
                if(Root != None)
                    Root.WindowEvent(WM_KeyDown, None, MouseX, MouseY, k);
                break;
            }
            break;
        case IST_Axis:
            switch (Key)
            {
            case IK_MouseX:
                MouseX = MouseX + (MouseScale * Delta);
                break;
            case IK_MouseY:
                MouseY = MouseY - (MouseScale * Delta);
                break;                  
            }
        default:
            break;
        }

        return true;
    }

Begin:
}

function LaunchUWindow()
{
    local int i;

    Root.GetPlayerOwner().Typing(true); //RUNE: Now typing

    Viewport.bSuspendPrecaching = True;
    bUWindowActive = !bQuickKeyEnable;
    Viewport.bShowWindowsMouse = True;

    if(bQuickKeyEnable)
        bNoDrawWorld = False;
    else
    {
        if(Viewport.Actor.Level.NetMode == NM_Standalone)
            Viewport.Actor.SetPause( True );
        bNoDrawWorld = ShowDesktop;
    }
    if(Root != None)
        Root.bWindowVisible = True;

    GotoState('UWindow');
}

function CloseUWindow()
{
    Root.GetPlayerOwner().Typing(false);    //RUNE: Not typing

    if(!bQuickKeyEnable)
        Viewport.Actor.SetPause( False );

    bNoDrawWorld = False;
    bQuickKeyEnable = False;
    bUWindowActive = False;
    Viewport.bShowWindowsMouse = False;

    if(Root != None)
        Root.bWindowVisible = False;
    GotoState('');
    Viewport.bSuspendPrecaching = False;
}

function CreateRootWindow(Canvas Canvas)
{
    local int i;

    if(Canvas != None)
    {
        OldClipX = Canvas.ClipX;
        OldClipY = Canvas.ClipY;
    }
    else
    {
        OldClipX = 0;
        OldClipY = 0;
    }
    
//	Log("Creating root window: "$RootWindow);
    
    Root = New(None) class<UWindowRootWindow>(DynamicLoadObject(RootWindow, class'Class'));

    Root.BeginPlay();
    Root.WinTop = 0;
    Root.WinLeft = 0;

    if(Canvas != None)
    {
        Root.WinWidth = Canvas.ClipX / Root.GUIScale;
        Root.WinHeight = Canvas.ClipY / Root.GUIScale;
        Root.RealWidth = Canvas.ClipX;
        Root.RealHeight = Canvas.ClipY;
    }
    else
    {
        Root.WinWidth = 0;
        Root.WinHeight = 0;
        Root.RealWidth = 0;
        Root.RealHeight = 0;
    }

    Root.ClippingRegion.X = 0;
    Root.ClippingRegion.Y = 0;
    Root.ClippingRegion.W = Root.WinWidth;
    Root.ClippingRegion.H = Root.WinHeight;

    Root.Console = Self;

    Root.bUWindowActive = bUWindowActive;

    Root.Created();
    bCreatedRoot = True;

    // Create the console window.
    ConsoleWindow = UWindowConsoleWindow(Root.CreateWindow(ConsoleClass, 100, 100, 200, 200));
    if(!bShowConsole)
        HideConsole();

    UWindowConsoleClientWindow(ConsoleWindow.ClientArea).TextArea.AddText(" ");
    for (I=0; I<4; I++)
        UWindowConsoleClientWindow(ConsoleWindow.ClientArea).TextArea.AddText(MsgText[I]);
}

function RenderUWindow( canvas Canvas )
{
    local UWindowWindow NewFocusWindow;

    Canvas.bNoSmooth = True;
    Canvas.Z = 1;
    Canvas.Style = 1;
    Canvas.DrawColor.r = 255;
    Canvas.DrawColor.g = 255;
    Canvas.DrawColor.b = 255;

    if(Viewport.bWindowsMouseAvailable && Root != None)
    {
        MouseX = Viewport.WindowsMouseX/Root.GUIScale;
        MouseY = Viewport.WindowsMouseY/Root.GUIScale;
    }

    if(!bCreatedRoot) 
        CreateRootWindow(Canvas);

    Root.bWindowVisible = True;
    Root.bUWindowActive = bUWindowActive;
    Root.bQuickKeyEnable = bQuickKeyEnable;

    if(Canvas.ClipX != OldClipX || Canvas.ClipY != OldClipY)
    {
        OldClipX = Canvas.ClipX;
        OldClipY = Canvas.ClipY;

        // RUNE: Explicitly set scale when changing resolution
        Root.ComputeGuiScale(Canvas.ClipX, Canvas.ClipY);
        
        Root.WinTop = 0;
        Root.WinLeft = 0;
        Root.WinWidth = Canvas.ClipX / Root.GUIScale;
        Root.WinHeight = Canvas.ClipY / Root.GUIScale;

        Root.RealWidth = Canvas.ClipX;
        Root.RealHeight = Canvas.ClipY;

        Root.ClippingRegion.X = 0;
        Root.ClippingRegion.Y = 0;
        Root.ClippingRegion.W = Root.WinWidth;
        Root.ClippingRegion.H = Root.WinHeight;

        Root.Resized();
    }

    if(MouseX > Root.WinWidth) MouseX = Root.WinWidth;
    if(MouseY > Root.WinHeight) MouseY = Root.WinHeight;
    if(MouseX < 0) MouseX = 0;
    if(MouseY < 0) MouseY = 0;


    // Check for keyboard focus
    NewFocusWindow = Root.CheckKeyFocusWindow();

    if(NewFocusWindow != Root.KeyFocusWindow)
    {
        Root.KeyFocusWindow.KeyFocusExit();     
        Root.KeyFocusWindow = NewFocusWindow;
        Root.KeyFocusWindow.KeyFocusEnter();
    }


    Root.MoveMouse(MouseX, MouseY);
    Root.WindowEvent(WM_Paint, Canvas, MouseX, MouseY, 0);
    if(bUWindowActive || bQuickKeyEnable) 
        Root.DrawMouse(Canvas);
}

event Message( PlayerReplicationInfo PRI, coerce string Msg, name N )
{
    local string OutText;

    Super.Message( PRI, Msg, N );

    if ( Viewport.Actor == None )
        return;

    if( Msg!="" )
    {
        if (( MsgType[TopLine] == 'Say' ) || ( MsgType[TopLine] == 'TeamSay' ))
            OutText = MsgPlayer[TopLine].PlayerName$": "$MsgText[TopLine];
        else
            OutText = MsgText[TopLine];
        if (ConsoleWindow != None)
            UWindowConsoleClientWindow(ConsoleWindow.ClientArea).TextArea.AddText(OutText);
    }
}

            UWindowConsoleClientWindow(ConsoleWindow.ClientArea).TextArea.AddText(Msg);
    }
}

function UpdateHistory()
{
    // Update history buffer.
    History[HistoryCur++ % MaxHistory] = TypedStr;
    if( HistoryCur > HistoryBot )
        HistoryBot++;
    if( HistoryCur - HistoryTop >= MaxHistory )
        HistoryTop = HistoryCur - MaxHistory + 1;
}

function HistoryUp()
{
    if( HistoryCur > HistoryTop )
    {
        History[HistoryCur % MaxHistory] = TypedStr;
        TypedStr = History[--HistoryCur % MaxHistory];
    }
}

function HistoryDown()
{
    History[HistoryCur % MaxHistory] = TypedStr;
    if( HistoryCur < HistoryBot )
        TypedStr = History[++HistoryCur % MaxHistory];
    else
        TypedStr="";
}

}
