class RuneServerAdminSpectator extends MessagingSpectator
    config;

struct PlayerMessage
{
    var PlayerReplicationInfo   PRI;
    var String                  Text;
    var Name                    Type;
    var PlayerMessage           Next;   // pointer to next message
};

var ListItem MessageList;

var byte ReceivedMsgNum;
var config byte ReceivedMsgMax;

var config bool bClientMessages;
var config bool bTeamMessages;
var config bool bVoiceMessages;
var config bool bLocalizedMessages;

function PreBeginPlay()
{
    Super.PreBeginPlay();
    bIsPlayer = False;
}

function AddMessage(PlayerReplicationInfo PRI, String S, name Type)
{
    local ListItem TempMsg;
    
    TempMsg = new(None) class'ListItem';
    TempMsg.Data = FormatMessage(PRI, S, Type);
    
    if (MessageList == None)
        MessageList = TempMsg;
    else
        MessageList.AddElement(TempMsg);
        
    if ((ReceivedMsgNum++) >= ReceivedMsgMax)
        MessageList.DeleteElement(MessageList); // delete the first element
}

    
function String FormatMessage(PlayerReplicationInfo PRI, String Text, name Type)
{
    local String Message, PName;
        
    // format Say and TeamSay messages
    if (PRI != None)
    {
       PName = PRI.PlayerName;
       if (Type == 'Say')
          Message = addColors(PRI.Team)$"<b>"$HTMLFormat(PName)$":</b></font><font color=#C0C0C0> "$HTMLFormat(Text)$"</font>";
       else if (Type == 'TeamSay')
          Message = addColors(PRI.Team)$"<b>["$HTMLFormat(PName)$"]:</b> "$HTMLFormat(Text)$"</font>";
       else
          Message = "("$Type$") "$HTMLFormat(Text)$"    "$PRI.Class;
    }
    else if (Type == 'Console')
        Message = HTMLFormat(Text);
    else if (Type == 'Event' && Left(Text,10) == "WebAdmin: ")
        Message = "<font color=#FFFFFF><b>WebAdmin:</b> "$HTMLFormat(Right(Text, (Len(Text) - 10)))$"</font>";
    else
        Message = "("$Type$") "$HTMLFormat(Text);
        
    return Message;
}

//The code below fixes how messages are a displayed in an HTML enviroment -Lar-
function String HTMLFormat(string text)
{
   ReplaceText(text, "&", "&");
   ReplaceText(text, ">", ">");
   ReplaceText(text, "<", "<");

   return text;

}

function String addColors(int i)
{
        if (i == 0) return "<font color=#AA0000>";
        if (i == 1) return "<font color=#0000AA>";
        if (i == 2) return "<font color=#008000>";
        if (i == 3) return "<font color=#BBBB00>";
        return "<font color=#FFFFFF>";
}

function ClientMessage( coerce string S, optional name Type, optional bool bBeep )
{
    if (bClientMessages && S != "")
        AddMessage(None, S, Type); 
}

function TeamMessage( PlayerReplicationInfo PRI, coerce string S, name Type, optional bool bBeep )
{
    if (bTeamMessages && S != "")
        AddMessage(PRI, S, Type);
}

function ClientVoiceMessage(PlayerReplicationInfo Sender, PlayerReplicationInfo Recipient, name messagetype, byte messageID)
{
    // do nothing?
}

function ReceiveLocalizedMessage( class<LocalMessage> Message, optional int Switch, optional PlayerReplicationInfo RelatedPRI_1, optional PlayerReplicationInfo RelatedPRI_2, optional Object OptionalObject )
{
    // do nothing?
}

defaultproperties
{
     ReceivedMsgMax=32
     bClientMessages=True
     bTeamMessages=True
     bLocalizedMessages=True
}
