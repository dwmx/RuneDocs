class RuneMenuSettingsBase extends RuneMenuPageWindow;

var RuneMenuServerClientWindow ServerParent;

var bool Initialized;
var float ControlOffset;

// Game Style
var UWindowComboControl StyleCombo;
var localized string StyleText;
var localized string Styles[3];
var localized string StyleHelp;

// Game Speed
var UWindowHSliderControl SpeedSlider;
var localized string SpeedText;
var localized string SpeedHelp;

// Autopickup (disabled until it works)
//var UWindowCheckbox AutopickupCheck;
//var localized string AutopickupText;
//var localized string AutopickupHelp;

// Retro
var UWindowCheckbox RetroMovementCheck;
var UWindowCheckbox RetroShieldSpeedCheck;
var UWindowCheckbox RetroThrownWeaponNoBlockCheck;
var UWindowCheckbox RetroWeaponDamageCheck;
var UWindowCheckbox RetroWeaponPowerupsCheck;
var UWindowCheckbox AdvancedThrowingCheck;
var UWindowCheckbox ChatLoggerCheck;

function Created()
{
    local int ControlWidth, ControlLeft, ControlRight;
    local int CenterWidth, CenterPos, ButtonWidth, ButtonLeft;

    Super.Created();

    ControlWidth = WinWidth/2.5;
    ControlLeft = (WinWidth/2 - ControlWidth)/2;
    ControlRight = WinWidth/2 + ControlLeft;

    CenterWidth = (WinWidth/4)*3;
    CenterPos = (WinWidth - CenterWidth)/2;

    ButtonWidth = WinWidth - 140;
    ButtonLeft = WinWidth - ButtonWidth - 40;

    ServerParent = RuneMenuServerClientWindow(GetParent(class'RuneMenuServerClientWindow'));
    if (ServerParent == None)
        Log("Error: RuneMenuSettingsBase without RuneMenuServerClientWindow parent.");

    // Game Style
    StyleCombo = UWindowComboControl(CreateControl(class'UWindowComboControl', CenterPos, ControlOffset, CenterWidth, 1));
    StyleCombo.SetText(StyleText);
    StyleCombo.SetHelpText(StyleHelp);
    StyleCombo.SetFont(F_Normal);
    StyleCombo.SetEditable(False);
    StyleCombo.AddItem(Styles[0]);
    StyleCombo.AddItem(Styles[1]);
    StyleCombo.AddItem(Styles[2]);
    ControlOffset += 25;

    // Game Speed
    SpeedSlider = UWindowHSliderControl(CreateControl(class'UWindowHSliderControl', CenterPos, ControlOffset, CenterWidth, 1));
    SpeedSlider.SetRange(50, 200, 5);
    SpeedSlider.SetHelpText(SpeedHelp);
    SpeedSlider.SetFont(F_Normal);
    ControlOffset += 25;

    // RetroMovement
    RetroMovementCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    RetroMovementCheck.SetText("1.08: Retro movement");
    RetroMovementCheck.SetHelpText("1.08: Retro movement");
    RetroMovementCheck.SetFont(F_Normal);
    RetroMovementCheck.bChecked = class'Engine.ServerOptions'.default.RetroMovement;
    RetroMovementCheck.Align = TA_Left;
    ControlOffset += 25;

    // RetroShieldSpeed
    RetroShieldSpeedCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    RetroShieldSpeedCheck.SetText("1.08: Retro shield behaviour");
    RetroShieldSpeedCheck.SetHelpText("1.08: Retro shield behaviour");
    RetroShieldSpeedCheck.SetFont(F_Normal);
    RetroShieldSpeedCheck.bChecked = class'Engine.ServerOptions'.Default.RetroShieldSpeed;
    RetroShieldSpeedCheck.Align = TA_Left;
    ControlOffset += 25;

    // RetroThrownWeaponNoBlock
    RetroThrownWeaponNoBlockCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    RetroThrownWeaponNoBlockCheck.SetText("1.08: Weapons don't block throws");
    RetroThrownWeaponNoBlockCheck.SetHelpText("1.08: Weapons don't block throws");
    RetroThrownWeaponNoBlockCheck.SetFont(F_Normal);
    RetroThrownWeaponNoBlockCheck.bChecked = class'Engine.ServerOptions'.Default.RetroThrownWeaponNoBlock;
    RetroThrownWeaponNoBlockCheck.Align = TA_Left;
    ControlOffset += 25;

    // RetroWeaponDamage
    RetroWeaponDamageCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    RetroWeaponDamageCheck.SetText("1.08: Retro weapon behaviour");
    RetroWeaponDamageCheck.SetHelpText("1.08: Retro weapon behaviour");
    RetroWeaponDamageCheck.SetFont(F_Normal);
    RetroWeaponDamageCheck.bChecked = class'Engine.ServerOptions'.Default.RetroWeaponDamage;
    RetroWeaponDamageCheck.Align = TA_Left;
    ControlOffset += 25;

    // RetroWeaponPowerups
    RetroWeaponPowerupsCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    RetroWeaponPowerupsCheck.SetText("1.08: Retro powerup behaviour");
    RetroWeaponPowerupsCheck.SetHelpText("1.08: Retro powerup behaviour");
    RetroWeaponPowerupsCheck.SetFont(F_Normal);
    RetroWeaponPowerupsCheck.bChecked = class'Engine.ServerOptions'.default.RetroWeaponPowerups;
    RetroWeaponPowerupsCheck.Align = TA_Left;
    ControlOffset += 25;

    // AdvancedThrowing
    AdvancedThrowingCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    AdvancedThrowingCheck.SetText("1.08: Advanced throwing");
    AdvancedThrowingCheck.SetHelpText("1.08: Advanced throwing");
    AdvancedThrowingCheck.SetFont(F_Normal);
    AdvancedThrowingCheck.bChecked = class'Engine.ServerOptions'.default.AdvancedThrowing;
    AdvancedThrowingCheck.Align = TA_Left;
    ControlOffset += 25;

    // AdvancedThrowing ControlRight
    ChatLoggerCheck = UWindowCheckbox(CreateControl(class'UWindowCheckbox', CenterPos, ControlOffset, CenterWidth, 1));
    ChatLoggerCheck.SetText("1.08: Chat Logging");
    ChatLoggerCheck.SetHelpText("1.08: Chat Logging");
    ChatLoggerCheck.SetFont(F_Normal);
    ChatLoggerCheck.bChecked = class'Engine.ServerOptions'.default.ChatLogger;
    ChatLoggerCheck.Align = TA_Left;
    ControlOffset += 25;

}

function SaveConfigs()
{
    class'Engine.ServerOptions'.static.StaticSaveConfig();
    Super.SaveConfigs();
}

function AfterCreate()
{
    Super.AfterCreate();

    DesiredWidth = 270;
    DesiredHeight = ControlOffset;

    LoadCurrentValues();
    Initialized = True;
}

function LoadCurrentValues()
{
}

function BeforePaint(Canvas C, float X, float Y)
{
    local int ControlWidth, ControlLeft, ControlRight;
    local int CenterWidth, CenterPos, ButtonWidth, ButtonLeft;

    Super.BeforePaint(C, X, Y);

    ControlWidth = WinWidth/2.5;
    ControlLeft = (WinWidth/2 - ControlWidth)/2;
    ControlRight = WinWidth/2 + ControlLeft;

    CenterWidth = (WinWidth/4)*3;
    CenterPos = (WinWidth - CenterWidth)/2;

    StyleCombo.SetSize(CenterWidth, 1);
    StyleCombo.WinLeft = CenterPos;
    StyleCombo.EditBoxWidth = 100;

    SpeedSlider.SetSize(CenterWidth, 1);
    SpeedSlider.SliderWidth = 100;
    SpeedSlider.WinLeft = CenterPos;

//	AutopickupCheck.SetSize(CenterWidth, 1);
//	AutopickupCheck.WinLeft = CenterPos;

    RetroMovementCheck.SetSize(ControlWidth, 1);
    RetroMovementCheck.WinLeft = CenterPos;

    RetroShieldSpeedCheck.SetSize(ControlWidth, 1);
    RetroShieldSpeedCheck.WinLeft = CenterPos;

    RetroThrownWeaponNoBlockCheck.SetSize(ControlWidth, 1);
    RetroThrownWeaponNoBlockCheck.WinLeft = CenterPos;

    RetroWeaponDamageCheck.SetSize(ControlWidth, 1);
    RetroWeaponDamageCheck.WinLeft = CenterPos;

    RetroWeaponPowerupsCheck.SetSize(ControlWidth, 1);
    RetroWeaponPowerupsCheck.WinLeft = CenterPos;

    AdvancedThrowingCheck.SetSize(ControlWidth, 1);
    AdvancedThrowingCheck.WinLeft = CenterPos;

    ChatLoggerCheck.SetSize(ControlWidth, 1);
    ChatLoggerCheck.WinLeft = CenterPos;
}

function Notify(UWindowDialogControl C, byte E)
{
    if (!Initialized)
        return;

    Super.Notify(C, E);

    switch(E)
    {
    case DE_Change:
        switch(C)
        {
            case StyleCombo:
                StyleChanged();
                break;
            case SpeedSlider:
                SpeedChanged();
                break;
            case RetroMovementCheck:
                RetroMovementChanged();
                break;
            case RetroShieldSpeedCheck:
                RetroShieldSpeedChanged();
                break;
            case RetroThrownWeaponNoBlockCheck:
                RetroThrownWeaponNoBlockChanged();
                break;
            case RetroWeaponDamageCheck:
                RetroWeaponDamageChanged();
                break;
            case RetroWeaponPowerupsCheck:
                RetroWeaponPowerupsChanged();
                break;
            case AdvancedThrowingCheck:
                AdvancedThrowingChanged();
                break;
            case ChatLoggerCheck:
                ChatLoggerChanged();
                break;
        }
    }
}
