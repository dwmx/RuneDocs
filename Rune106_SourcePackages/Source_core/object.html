function string GetToken(out string s)
{
    local int i,length;
    local string tok;

    if( instr(s," ") != -1 )
    {
        length = Len(s);
        for (i=0; i<length; i++)
        {
            if (Mid(s, i, 1) == " ")
            {
                tok = left(s, i);
                s = right(s, length-i-1);
                break;
            }
        }
    }
    else
    {
        if (tok=="")
        {
            tok = s;
            s = "";
        }
    }

    return tok;
}
